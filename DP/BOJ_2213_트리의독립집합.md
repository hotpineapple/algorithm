# 트리의 독립집합
- 트리에서의 DP
- 사이클이 없는 그래프라는 점이 힌트
- 방문 처리에 유의
- 코드
```java
import java.io.*;
import java.util.*;

public class Main {
    static int N, W[];
    static List<List<Integer>> list;
    static boolean vst[];
    static Data[][] D;
    static class Data {
        int sum;
        List<Integer> list = new ArrayList<>();
    }
    public static void main(String[] args) throws IOException {
        // Input
        System.setIn(new FileInputStream("src/a.txt"));
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        N = Integer.parseInt(br.readLine());

        StringTokenizer st = new StringTokenizer(br.readLine());
        W = new int[N+1];
        vst = new boolean[N+1];
        for(int i=1;i<=N;i++) W[i] = Integer.parseInt(st.nextToken());

        list = new ArrayList<>();
        for(int i=1;i<=N+1;i++) list.add(new ArrayList<>());
        String line;
        while((line = br.readLine())!=null){
            st = new StringTokenizer(line);
            int u = Integer.parseInt(st.nextToken());
            int v = Integer.parseInt(st.nextToken());
            list.get(u).add(v);
            list.get(v).add(u);
        }

        // DP
        D = new Data[N+1][2];
        dfs(1);

        // output
        if(D[1][0].sum>D[1][1].sum){
            System.out.println(D[1][0].sum);
            Collections.sort(D[1][0].list);
            for(int i=0;i<D[1][0].list.size();i++){
                System.out.print(D[1][0].list.get(i)+" ");
            }
        }else {
            System.out.println(D[1][1].sum);
            Collections.sort(D[1][1].list);
            for(int i=0;i<D[1][1].list.size();i++){
                System.out.print(D[1][1].list.get(i)+ " ");
            }
        }
    }
    private static void dfs(int node) {
        vst[node] = true;
        D[node][0] = new Data();
        D[node][1] = new Data();

        if(node == N) {
            D[node][0].sum = W[node];
            D[node][0].list.add(node);

            return;
        }
        List<Integer> a = list.get(node);
        for(int i=0;i<a.size();i++) {
            int v = a.get(i);
            if(!vst[v]) {
                // 연결된 노드
                dfs(v);

                // 이 정점 선택
                D[node][0].sum += D[v][1].sum;
                for(int j=0;j<D[v][1].list.size();j++){
                    D[node][0].list.add(D[v][1].list.get(j));
                }

                // 이 정점 선택 안함
                if(D[v][0].sum>D[v][1].sum) {
                    D[node][1].sum += D[v][0].sum;
                    for(int j=0;j<D[v][0].list.size();j++) {
                        D[node][1].list.add(D[v][0].list.get(j));
                    }
                }else {
                    D[node][1].sum += D[v][1].sum;
                    for(int j=0;j<D[v][1].list.size();j++) {
                        D[node][1].list.add(D[v][1].list.get(j));
                    }
                }
            }
        }
        D[node][0].sum += W[node];
        D[node][0].list.add(node);
    }
}
```
