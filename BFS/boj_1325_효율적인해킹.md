# 효율적인 해킹
- 신뢰관계(=해킹 가능)가 주어졌을 때 한번에 가장 많은 컴퓨터를 해킹할 수 있는 컴퓨터번호를 구하는 문제
- BFS 를 이용하여 함께 해킹할 수 있는 개수를 더해주었다.
- 코드
```java
import java.io.*;
import java.util.*;

public class Main {
    public static void main(String[] args) throws Exception {
        // Input
        // System.setIn(new FileInputStream("src/a.txt"));
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st = new StringTokenizer(br.readLine());
        int N = Integer.parseInt(st.nextToken());
        int M = Integer.parseInt(st.nextToken());
        List<Integer> adj[] = new ArrayList[N+1];
        for(int i=1;i<=N;i++) adj[i] = new ArrayList<>();
        boolean[] vst;
        for(int i=0;i<M;i++) {
            st = new StringTokenizer(br.readLine());
            int a = Integer.parseInt(st.nextToken());
            int b = Integer.parseInt(st.nextToken());
            adj[b].add(a);
        }
        int max = 0;
        List<Integer> ans = new ArrayList<>();
        for(int i=1;i<=N;i++) {
            vst = new boolean[N+1];
            vst[i]=true;
            int cnt = 1;
            // bfs
            Queue<Integer> q = new LinkedList<>();
            List<Integer> list = adj[i];
            for (int j=0;j<list.size();j++) {
                if (vst[list.get(j)]) {
                    continue;
                }
                vst[list.get(j)]= true;
                q.offer(list.get(j));
            }
            while(!q.isEmpty()){
                int node = q.poll();
                cnt++;
                List<Integer> list2 = adj[node];
                for (int j=0;j<list2.size();j++) {
                    if (!vst[list2.get(j)]) {
                        vst[list2.get(j)] = true;
                        q.offer(list2.get(j));
                    }
                }
            }

            // max cmp
            if(max<cnt){
                max=cnt;
                ans.clear();
                ans.add(i);
            }else if(max==cnt){
                ans.add(i);
            }
        }

        for(int i=0;i<ans.size();i++) {
            System.out.print(ans.get(i)+" ");
        }
    }
}
```
